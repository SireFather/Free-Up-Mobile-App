<script>
    // --- Canvas and Context Setup ---
    const canvas = document.getElementById('gameCanvas');
    const ctx = canvas.getContext('2d');
    
    // --- Game State Variables ---
    let gameLoopId;
    let score = 0;
    let lives = 3;
    let gameOver = false;
    let lastAsteroidSpawn = 0;
    const ASTEROID_SPAWN_INTERVAL = 1500; // ms
    const ASTEROID_MAX_SPEED = 2;
    const ASTEROID_MIN_SPEED = 0.5;
    
    // --- Entities Arrays ---
    let player = {};
    let bullets = [];
    let asteroids = [];
    let particles = [];
    
    // --- Input States (Keyboard and Touch) ---
    const input = {
        left: false,
        right: false,
        fire: false
    };

    // --- Utility Functions ---

    /** Simple distance calculation */
    function distance(x1, y1, x2, y2) {
        return Math.sqrt(Math.pow(x2 - x1, 2) + Math.pow(y2 - y1, 2));
    }

    /** Updates the UI displays */
    function updateUI() {
        document.getElementById('score-display').textContent = score;
        document.getElementById('lives-display').textContent = lives;
    }
    
    /** Shows the Game Over screen */
    function showGameOver() {
        gameOver = true;
        cancelAnimationFrame(gameLoopId);
        document.getElementById('final-score').textContent = score;
        document.getElementById('game-over-modal').style.display = 'flex';
    }

    // --- Entity Classes (Simplified Objects) ---

    // Player (Ship)
    function Player() {
        this.x = canvas.width / 2;
        this.y = canvas.height - 40;
        this.size = 15;
        this.speed = 5;
        this.color = '#38bdf8'; // Sky Blue
        this.canFire = true;
        this.fireRate = 200; // ms
        
        this.draw = function() {
            ctx.fillStyle = this.color;
            ctx.beginPath();
            // Nose of the ship
            ctx.moveTo(this.x, this.y - this.size);
            // Bottom left
            ctx.lineTo(this.x - this.size, this.y + this.size);
            // Bottom right
            ctx.lineTo(this.x + this.size, this.y + this.size);
            ctx.closePath();
            ctx.fill();

            // Simple engine light
            ctx.fillStyle = '#fcd34d'; // Yellow
            ctx.fillRect(this.x - 5, this.y + this.size, 10, 3);
        };
        
        this.update = function() {
            if (input.left) {
                this.x = Math.max(this.size, this.x - this.speed);
            }
            if (input.right) {
                this.x = Math.min(canvas.width - this.size, this.x + this.speed);
            }
            if (input.fire && this.canFire) {
                this.shoot();
            }
        };

        this.shoot = function() {
            // Add a new bullet
            bullets.push(new Bullet(this.x, this.y - this.size));
            this.canFire = false;
            
            // Set a timeout to allow firing again
            setTimeout(() => {
                this.canFire = true;
            }, this.fireRate);
        };
    }

    // Bullet
    function Bullet(x, y) {
        this.x = x;
        this.y = y;
        this.radius = 3;
        this.speed = 7;
        this.color = '#10b981'; // Green
        
        this.draw = function() {
            ctx.fillStyle = this.color;
            ctx.beginPath();
            ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2);
            ctx.fill();
        };
        
        this.update = function() {
            this.y -= this.speed;
        };
    }

    // Asteroid
    function Asteroid(x, y, radius, vx, vy) {
        this.x = x;
        this.y = y;
        this.radius = radius;
        this.vx = vx;
        this.vy = vy;
        this.color = '#6b7280'; // Gray
        
        this.draw = function() {
            ctx.strokeStyle = this.color;
            ctx.lineWidth = 2;
            ctx.beginPath();
            // Draw a slightly irregular circle (for that asteroid look)
            let complexity = 0.8;
            let numPoints = 8;
            for(let i = 0; i < numPoints; i++) {
                let angle = (i / numPoints) * Math.PI * 2;
                let r = this.radius + (Math.random() - 0.5) * complexity * this.radius;
                let px = this.x + r * Math.cos(angle);
                let py = this.y + r * Math.sin(angle);
                if (i === 0) {
                    ctx.moveTo(px, py);
                } else {
                    ctx.lineTo(px, py);
                }
            }
            ctx.closePath();
            ctx.stroke();
        };
        
        this.update = function() {
            this.x += this.vx;
            this.y += this.vy;
        };
    }

    // Particle for explosions
    function Particle(x, y, color, size, vx, vy, lifespan) {
        this.x = x;
        this.y = y;
        this.color = color;
        this.size = size;
        this.vx = vx;
        this.vy = vy;
        this.lifespan = lifespan;
        this.age = 0;

        this.draw = function() {
            ctx.fillStyle = this.color;
            ctx.globalAlpha = 1 - (this.age / this.lifespan); // Fade out
            ctx.fillRect(this.x, this.y, this.size, this.size);
            ctx.globalAlpha = 1;
        };

        this.update = function() {
            this.x += this.vx;
            this.y += this.vy;
            this.age += 1;
        };
    }


    // --- Game Logic Functions ---

    /** Initializes all game entities and state */
    function initGame() {
        // Reset state
        score = 0;
        lives = 3;
        gameOver = false;
        bullets = [];
        asteroids = [];
        particles = [];
        lastAsteroidSpawn = performance.now();
        
        // Create player
        player = new Player();
        player.x = canvas.width / 2;
        player.y = canvas.height - 40;

        // Reset UI and hide modal
        updateUI();
        document.getElementById('game-over-modal').style.display = 'none';

        // Start loop
        if (gameLoopId) cancelAnimationFrame(gameLoopId);
        gameLoopId = requestAnimationFrame(gameLoop);
    }

    /** Spawns a new asteroid from a random edge */
    function spawnAsteroid(timestamp) {
        if (timestamp - lastAsteroidSpawn < ASTEROID_SPAWN_INTERVAL) return;

        lastAsteroidSpawn = timestamp;
        
        const radius = Math.random() * 20 + 10; // 10 to 30
        const speed = Math.random() * (ASTEROID_MAX_SPEED - ASTEROID_MIN_SPEED) + ASTEROID_MIN_SPEED;

        let x, y, vx, vy;
        
        // Randomly choose edge to spawn from (0: top, 1: left, 2: right)
        const edge = Math.floor(Math.random() * 3);

        if (edge === 0) { // Top edge
            x = Math.random() * canvas.width;
            y = -radius;
            // Target the bottom half of the screen
            const angle = Math.atan2(Math.random() * canvas.height * 0.5 + canvas.height * 0.5 - y, Math.random() * canvas.width - x);
            vx = speed * Math.cos(angle);
            vy = speed * Math.sin(angle);
        } else if (edge === 1) { // Left edge
            x = -radius;
            y = Math.random() * canvas.height / 2;
            // Target the center-right area
            const angle = Math.atan2(Math.random() * canvas.height - y, canvas.width / 2 + Math.random() * canvas.width / 2 - x);
            vx = speed * Math.cos(angle);
            vy = speed * Math.sin(angle);
        } else { // Right edge
            x = canvas.width + radius;
            y = Math.random() * canvas.height / 2;
            // Target the center-left area
            const angle = Math.atan2(Math.random() * canvas.height - y, canvas.width / 2 - Math.random() * canvas.width / 2 - x);
            vx = speed * Math.cos(angle);
            vy = speed * Math.sin(angle);
        }

        asteroids.push(new Asteroid(x, y, radius, vx, vy));
    }

    /** Creates a simple explosion of particles */
    function createExplosion(x, y, count, color) {
        for (let i = 0; i < count; i++) {
            const angle = Math.random() * Math.PI * 2;
            const speed = Math.random() * 3 + 1;
            const vx = speed * Math.cos(angle);
            const vy = speed * Math.sin(angle);
            const size = Math.random() * 2 + 1;
            const lifespan = Math.random() * 40 + 20;
            particles.push(new Particle(x, y, color, size, vx, vy, lifespan));
        }
    }

    /** Checks and handles all collisions */
    function checkCollisions() {
        // Bullet vs Asteroid collision
        for (let i = bullets.length - 1; i >= 0; i--) {
            const bullet = bullets[i];
            
            for (let j = asteroids.length - 1; j >= 0; j--) {
                const asteroid = asteroids[j];
                
                if (distance(bullet.x, bullet.y, asteroid.x, asteroid.y) < asteroid.radius + bullet.radius) {
                    // Collision occurred!
                    createExplosion(asteroid.x, asteroid.y, asteroid.radius * 2, '#fcd34d'); // Yellow explosion
                    
                    // Remove both
                    bullets.splice(i, 1);
                    asteroids.splice(j, 1);
                    
                    score += 10;
                    updateUI();
                    break; // Bullet is destroyed, move to next bullet
                }
            }
        }

        // Player vs Asteroid collision (simplified bounding box for player)
        for (let i = asteroids.length - 1; i >= 0; i--) {
            const asteroid = asteroids[i];
            
            // Check if asteroid center is near player's bounding area
            const playerCenterY = player.y; // Center point for simplification
            const collisionDistance = player.size + asteroid.radius;

            if (distance(player.x, playerCenterY, asteroid.x, asteroid.y) < collisionDistance) {
                // Collision occurred!
                createExplosion(player.x, player.y, 40, '#f87171'); // Red explosion
                createExplosion(asteroid.x, asteroid.y, asteroid.radius * 3, '#f87171');
                
                // Remove asteroid
                asteroids.splice(i, 1);
                
                lives--;
                updateUI();
                
                if (lives <= 0) {
                    showGameOver();
                    return;
                }
            }
        }
    }

    /** Main game update logic */
    function updateGame() {
        if (gameOver) return;

        // Update Player
        player.update();
        
        // Update Bullets
        for (let i = bullets.length - 1; i >= 0; i--) {
            bullets[i].update();
            // Remove off-screen bullets
            if (bullets[i].y < -bullets[i].radius) {
                bullets.splice(i, 1);
            }
        }
        
        // Update Asteroids
        for (let i = asteroids.length - 1; i >= 0; i--) {
            asteroids[i].update();
            // Remove off-screen asteroids (if they pass the bottom)
            if (asteroids[i].y > canvas.height + asteroids[i].radius) {
                asteroids.splice(i, 1);
                // Optionally penalize player for missed asteroid
            }
        }

        // Update Particles
        for (let i = particles.length - 1; i >= 0; i--) {
            particles[i].update();
            if (particles[i].age >= particles[i].lifespan) {
                particles.splice(i, 1);
            }
        }

        checkCollisions();
    }

    /** Main game drawing logic */
    function drawGame() {
        // Clear the canvas
        ctx.clearRect(0, 0, canvas.width, canvas.height);
        
        // Draw all entities
        player.draw();

        bullets.forEach(b => b.draw());
        asteroids.forEach(a => a.draw());
        particles.forEach(p => p.draw());
    }

    /** Main game loop function */
    function gameLoop(timestamp) {
        // Handle spawning
        spawnAsteroid(timestamp);

        // Update
        updateGame();

        // Draw
        drawGame();
        
        // Loop
        gameLoopId = requestAnimationFrame(gameLoop);
    }
    
    // --- Event Listeners ---
    
    // 1. Keyboard Controls (Desktop)
    document.addEventListener('keydown', (e) => {
        if (gameOver) return;
        if (e.key === 'ArrowLeft' || e.key === 'a' || e.key === 'A') input.left = true;
        if (e.key === 'ArrowRight' || e.key === 'd' || e.key === 'D') input.right = true;
        if (e.key === ' ' || e.key === 'w' || e.key === 'W') input.fire = true;
    });

    document.addEventListener('keyup', (e) => {
        if (e.key === 'ArrowLeft' || e.key === 'a' || e.key === 'A') input.left = false;
        if (e.key === 'ArrowRight' || e.key === 'd' || e.key === 'D') input.right = false;
        if (e.key === ' ' || e.key === 'w' || e.key === 'W') input.fire = false;
    });
    
    // 2. Mobile/Touch Controls
    const leftButton = document.getElementById('left-button');
    const rightButton = document.getElementById('right-button');
    const fireButton = document.getElementById('fire-button');
    const restartButton = document.getElementById('restart-button');

    // Generic touch handler for pressing/releasing a button
    function setupTouchControl(element, inputProperty) {
        // Press down/Start touch
        element.addEventListener('touchstart', (e) => {
            e.preventDefault();
            if (!gameOver) input[inputProperty] = true;
        });
        
        // Release/End touch
        element.addEventListener('touchend', (e) => {
            e.preventDefault();
            input[inputProperty] = false;
        });
        // Handle cancellation (like moving touch outside the element)
        element.addEventListener('touchcancel', (e) => {
            e.preventDefault();
            input[inputProperty] = false;
        });
    }

    setupTouchControl(leftButton, 'left');
    setupTouchControl(rightButton, 'right');
    setupTouchControl(fireButton, 'fire');

    // Restart Button
    restartButton.addEventListener('click', initGame);

    // 3. Canvas Resizing for Responsiveness
    function resizeCanvas() {
        const container = document.getElementById('game-container');
        // Set fixed aspect ratio (e.g., 600x400 ratio)
        const aspectRatio = 600 / 400;
        
        // Get the computed width of the container (max 600px)
        const containerWidth = container.clientWidth - 20; // accounting for container padding
        
        canvas.width = containerWidth;
        canvas.height = containerWidth / aspectRatio;

        // Re-center player on resize
        if (player.x) {
            player.x = canvas.width / 2;
        }
    }

    // --- Initialization ---
    window.addEventListener('load', () => {
        // Initial setup
        resizeCanvas();
        initGame();
    });
    
    window.addEventListener('resize', resizeCanvas);

</script>
